import { useState } from "react";

export default function StockReceivedModal({ onClose }) {
  const [tab, setTab] = useState<"rolls" | "meat" | "drinks">("rolls");

  return (
    <div className="modal">
      <div className="tabs">
        {["rolls", "meat", "drinks"].map((t) => (
          <button key={t} onClick={() => setTab(t as any)}>
            {t.toUpperCase()}
          </button>
        ))}
      </div>

      {tab === "rolls" && <RollsTab />}
      {tab === "meat" && <MeatTab />}
      {tab === "drinks" && <DrinksTab />}

      <button onClick={onClose}>Close</button>
    </div>
  );
}

function RollsTab() {
  const [qty, setQty] = useState(0);
  const [paid, setPaid] = useState(false);

  async function submit() {
    await fetch("/api/stock/rolls", {
      method: "POST",
      body: JSON.stringify({ qty, paid }),
    });
  }

  return (
    <>
      <input
        type="number"
        value={qty}
        onChange={(e) => setQty(Number(e.target.value))}
        placeholder="Rolls purchased"
      />

      <label>
        <input
          type="checkbox"
          checked={paid}
          onChange={() => setPaid(!paid)}
        />
        Paid at purchase (create expense)
      </label>

      <button onClick={submit}>Log Rolls</button>
    </>
  );
}

const DRINKS = [
  "Coke",
  "Coke Zero",
  "Sprite",
  "Fanta Orange",
  "Fanta Strawberry",
  "Soda Water",
  "Bottled Water",
  "Kids Juice (Apple)",
  "Kids Juice (Orange)",
];

function DrinksTab() {
  const [counts, setCounts] = useState(
    Object.fromEntries(DRINKS.map((d) => [d, 0]))
  );

  async function submit() {
    await fetch("/api/stock/drinks", {
      method: "POST",
      body: JSON.stringify({ counts }),
    });
  }

  return (
    <>
      {DRINKS.map((d) => (
        <div key={d}>
          <label>{d}</label>
          <input
            type="number"
            value={counts[d]}
            onChange={(e) =>
              setCounts({ ...counts, [d]: Number(e.target.value) })
            }
          />
        </div>
      ))}

      <button onClick={submit}>Log Drinks</button>
    </>
  );
}
